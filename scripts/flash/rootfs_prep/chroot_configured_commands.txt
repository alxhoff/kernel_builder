# Per-robot chroot config, run by setup_rootfs_as_robot_for_flashing.sh.
# Designed to be re-runnable (e.g. after a previous failed run leaves a
# half-configured package behind) and tolerant of bare-Ubuntu rootfs that
# never saw setup_tegra_package.sh's chroot phase.
echo "Recovering from any half-configured packages from previous runs"
dpkg --configure -a || true
apt-get update
DEBIAN_FRONTEND=noninteractive apt-get -y --fix-broken install || true
dpkg -r cartken-vpn-initial || true
dpkg -r cartken-vpn-initial-credentials || true
echo "Installing apt deps (covers cartken-* deb dependencies)"
DEBIAN_FRONTEND=noninteractive apt-get install -y libqmi-utils uhubctl python3-pip ifupdown
echo "Installing viki"
python3 -m pip install /root/packages/cartken-wheels/viki-*.whl
echo "Installing cartken debians via apt so missing deps resolve from Ubuntu repos"
DEBIAN_FRONTEND=noninteractive apt-get install -y /root/packages/cartken-jetson-debians/cartken-networking-base.deb
DEBIAN_FRONTEND=noninteractive apt-get install -y /root/packages/cartken-jetson-debians/cartken-modem-tools.deb
DEBIAN_FRONTEND=noninteractive apt-get install -y /root/packages/cartken-jetson-debians/cartken-jetson-modem-manager.deb
DEBIAN_FRONTEND=noninteractive apt-get install -y /root/packages/cartken-jetson-debians/cartken-routing-tables.deb
DEBIAN_FRONTEND=noninteractive apt-get install -y /root/packages/cartken-jetson-debians/cartken-modem-routing-table-manager.deb
DEBIAN_FRONTEND=noninteractive apt-get install -y /root/packages/cartken-jetson-debians/cartken-vpn-common-credentials.deb /root/packages/cartken-jetson-debians/cartken-vpn.deb
echo "Installing cartken-jetson-sshd-v2 (port 8612, cert-based auth)"
DEBIAN_FRONTEND=noninteractive apt-get install -y /root/packages/cartken-jetson-debians/cartken-jetson-sshd-v2.deb
