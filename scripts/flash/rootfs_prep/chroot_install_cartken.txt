# Cartken-layer chroot driver (purge + reinstall cartken-* packages).
#
# Which debs to install: cartken_jetson_debs.txt (one basename per line).
# That manifest is copied to /root/cartken_jetson_debs.txt before this runs.
# Runs as the second chroot pass in setup_tegra_package.sh (after
# chroot_install_os_jp{5,6}.txt has installed the OS deps), and again
# from setup_rootfs_as_robot_for_flashing.sh and setup_rootfs_as_robot_for_ota.sh
# whenever a per-robot rootfs is prepared (so a --tag swap re-installs from
# the freshly pulled debs).
#
# Re-runnable: starts each pass by purging *every* installed cartken-*
# package, then reinstalls the full intended set in a single transaction.
# The deb set itself lives in cartken_jetson_debs.txt (one basename per line);
# removing a package there will actually remove it from the rootfs the next
# time this file runs - no separate "uninstall" step needed.
echo "Recovering from any half-configured packages from previous runs"
dpkg --configure -a || true
apt-get update
DEBIAN_FRONTEND=noninteractive apt-get -y --fix-broken install || true
echo "Make /root/packages readable by apt sandbox user _apt"
[ -d /root/packages ] && chmod -R a+rX /root/packages
echo "Purging every cartken-* package so reinstall is from a clean slate"
if INSTALLED=$(dpkg-query -W -f='${Package}\n' 'cartken-*' 2>/dev/null) && [ -n "$INSTALLED" ]; then if ! DEBIAN_FRONTEND=noninteractive apt-get -y purge $INSTALLED; then echo "WARN: apt purge failed (often a cartken postrm bug under /bin/sh); forcing dpkg --purge --force-all"; for pkg in $INSTALLED; do dpkg --purge --force-all "$pkg" 2>/dev/null || true; done; dpkg --configure -a || true; fi; fi
echo "Defensive: ensure python3-pip + ifupdown exist (jp{5,6}.txt normally installs these, but this file is also re-runnable on rootfs that have not seen the OS layer recently)"
DEBIAN_FRONTEND=noninteractive apt-get install -y python3-pip ifupdown
echo "Installing viki (cartken python wheel)"
python3 -m pip install /root/packages/cartken-wheels/viki-*.whl
echo "Installing the full cartken-* set from /root/cartken_jetson_debs.txt in one apt invocation so dep order is auto-resolved"
[[ -f /root/cartken_jetson_debs.txt ]] || { echo "Missing /root/cartken_jetson_debs.txt (should be copied into rootfs before chroot)"; exit 1; }
_debs=(); while IFS= read -r pkg || [[ -n "$pkg" ]]; do [[ "$pkg" =~ ^[[:space:]]*# ]] && continue; pkg="${pkg%%#*}"; pkg="${pkg// /}"; [[ -z "$pkg" ]] && continue; _debs+=("/root/packages/cartken-jetson-debians/${pkg}.deb"); done < /root/cartken_jetson_debs.txt; [[ ${#_debs[@]} -gt 0 ]] || { echo "No packages listed in /root/cartken_jetson_debs.txt"; exit 1; }; DEBIAN_FRONTEND=noninteractive apt-get install -y "${_debs[@]}"
echo "Cleaning up unused dependencies + apt cache + dangling systemd symlinks"
DEBIAN_FRONTEND=noninteractive apt-get -y autoremove --purge || true
apt-get clean || true
find /etc/systemd/system -xtype l -delete 2>/dev/null || true
