# OS-layer chroot for JetPack 6.x rootfs (Pass 1/2; run by setup_tegra_package.sh).
# Installs the system apt packages cartken-* (and viki) depend on, holds the
# nvidia-l4t metapackages so apt upgrade does not move them, and configures
# stock /etc/ssh/sshd_config (legacy v1 sshd, being phased out by
# cartken-jetson-sshd-v2). All cartken-* / viki installation lives in
# chroot_install_cartken.txt and runs immediately after this file; do NOT add
# cartken-* installs here.
apt-get update -o Acquire::Retries=5 -o APT::Update::Error-Mode=any
apt-mark hold $(dpkg -l | grep 'nvidia-l4t-' | awk '{print $2}')
DEBIAN_FRONTEND=noninteractive apt-get upgrade -y || true
apt-mark unhold $(dpkg -l | grep 'nvidia-l4t-' | awk '{print $2}')
dpkg -l | awk '/^ii/ {print $2}' | grep -E '^nvidia-l4t-(kernel|kernel-headers|kernel-dtbs|firmware|xusb-firmware)$' | xargs -r apt-mark hold
apt-get autoremove -y
echo "Installing apt packages (single transaction)"
DEBIAN_FRONTEND=noninteractive apt-get install -y python3-pip ifupdown v4l-utils apt-utils tree fail2ban tmux vim screen htop can-utils iptraf libqt5core5a cmake libssl-dev libc6-dev build-essential libudev-dev libqmi-utils uhubctl ffmpeg libqt5widgets5 uuid-runtime nvme-cli docker.io nvidia-container-toolkit nvidia-jetpack-runtime chrony nano git python3-libnvinfer-dev
echo "Configuring NVIDIA container runtime for docker"
mkdir -p /etc/docker
nvidia-ctk runtime configure --runtime=docker --set-as-default
pip install -U sphinx python-can
sed -i 's/^#\?\(PermitRootLogin\).*/\1 yes/' /etc/ssh/sshd_config
echo "root:$(openssl passwd -6 cartken)" | chpasswd -e
echo "Cleaning up unused dependencies + apt cache + dangling systemd symlinks"
DEBIAN_FRONTEND=noninteractive apt-get -y autoremove --purge || true
apt-get clean || true
find /etc/systemd/system -xtype l -delete 2>/dev/null || true
